1//! Structures needed for working with QC-MDPC codes
2//!
3//! This module provides:
4//! - Core MDPC code parameters including:
5//!   - Block length (`r`)
6//!   - Code length (`n`)
7//!   - Column weight (`d`)
8//!   - Row weight (`w`)
9//! - Degree distributions of nodes in the subgraph `G` of the Tanner graph
10//!   given by near-codewords of the form `xⁱh0(x)⊕0` and `0⊕xⁱh1(x)` (see §2.3
11//!   of \[ALMPRTV25])
12//!
13//! ## Reference
14//!
15//! > \[ALMPRTV25]: Sarah Arpin, Jun Bo Lau, Antoine Mesnard, Ray Perlner, Angela Robinson, Jean-Pierre Tillich & Valentin Vasseur: Error floor prediction with Markov models for QC-MDPC codes. <https://eprint.iacr.org/2025/153>
16use std::{collections::HashMap, fmt::Debug};
17
18use crate::errors::{Error, Result};
19
20/// Structure for an MDPC code.
21#[derive(Clone, Debug)]
22pub struct MDPCCode {
23    /// Block length
24    pub r: usize,
25    /// Code length
26    pub n: usize,
27    /// Column weight of the parity-check matrix
28    pub d: usize,
29    /// Row weight of the parity-check matrix
30    pub w: usize,
31    /// Number of circulant blocks
32    pub index: usize,
33}
34
35/// Represents the degree distribution of parity check nodes in the subgraph `G`
36/// of the Tanner graph induced by near-codewords in a QC-MDPC code.
37///
38/// These near-codewords are of the form `ν=h0(x)⊕0` and `ν=0⊕h1(x)`.
39///
40/// # Notation
41///
42/// - `n_Δ`: number of parity-check equations in `G` of degree `Δ`
43#[derive(Debug)]
44pub struct CodeDegrees {
45    /// Degree distribution of parity check nodes
46    ///
47    /// `degrees[i][Δ] = n_i,Δ`
48    ///
49    /// where `n_i,Δ` is the number of parity-check equations in `G` of degree
50    /// `Δ` for block `i`
51    pub degrees: Vec<BlockDegrees>,
52}
53
54/// Degree distribution of a single block
55///
56/// Use a simple map `{ Δ: n_Δ }`
57/// where `n_Δ` is the number of parity-check equations in `G` of degree `Δ` for
58/// the given block
59pub type BlockDegrees = HashMap<usize, usize>;
60
61impl MDPCCode {
62    /// Creates a new `MDPCCode` instance with parameter validation
63    ///
64    /// # Arguments
65    /// * `r` - Block length
66    /// * `n` - Code length
67    /// * `d` - Column weight of the parity-check matrix
68    /// * `w` - Row weight of the parity-check matrix
69    /// * `index` - Number of circulant blocks
70    ///
71    /// # Errors
72    /// Returns `Error::Config` if parameters are invalid
73    pub fn new(r: usize, n: usize, d: usize, w: usize, index: usize) -> Result<Self> {
74        if d > r {
75            return Err(Error::config(format!(
76                "Block weight ({}) cannot exceed block length ({})",
77                d, r
78            )));
79        }
80        if w > n {
81            return Err(Error::config(format!(
82                "Row weight ({}) cannot exceed code length ({})",
83                w, n
84            )));
85        }
86        if index == 0 {
87            return Err(Error::config("Number of circulant blocks must be positive"));
88        }
89
90        Ok(MDPCCode { r, n, d, w, index })
91    }
92}
93
94impl CodeDegrees {
95    /// Creates a new `CodeDegrees` instance with the given degree
96    /// distributions.
97    ///
98    /// This function validates the degree distributions according to the
99    /// constraints of the QC-MDPC code structure. It checks that:
100    /// 1. the sum of all `count`s for odd degrees is `d`,
101    /// 2. the sum of all `count`s is `r`,
102    /// 3. the sum of all `degree * count` is `d²`.
103    ///
104    /// # Arguments
105    /// * `code` - A reference to the `MDPCCode`.
106    /// * `degrees` - An array of degree distributions of nodes in the subgraph
107    ///   `G`.
108    ///
109    /// # Returns
110    /// A `Result` containing either the validated `CodeDegrees` instance or an
111    /// error.
112    pub fn new(code: &MDPCCode, degrees: Vec<HashMap<usize, usize>>) -> Result<Self> {
113        for (block_idx, block_degrees) in degrees.iter().enumerate() {
114            let sum_odd: usize = block_degrees
115                .iter()
116                .filter(|&(&k, _)| k % 2 == 1)
117                .map(|(_, &v)| v)
118                .sum();
119            if sum_odd != code.d {
120                return Err(Error::config(format!(
121                    "Degrees for block {}: sum of counts for odd degrees ({}) must equal d ({})",
122                    block_idx, sum_odd, code.d
123                )));
124            }
125
126            let sum_all: usize = block_degrees.values().sum();
127            if sum_all != code.r {
128                return Err(Error::config(format!(
129                    "Degrees for block {}: sum of all counts ({}) must equal r ({})",
130                    block_idx, sum_all, code.r
131                )));
132            }
133
134            let sum_degree_count: usize = block_degrees.iter().map(|(&k, &v)| k * v).sum();
135            if sum_degree_count != code.d.pow(2) {
136                return Err(Error::config(format!(
137                    "Degrees for block {}: sum of degree*count ({}) must equal d² ({})",
138                    block_idx,
139                    sum_degree_count,
140                    code.d.pow(2)
141                )));
142            }
143        }
144
145        Ok(CodeDegrees { degrees })
146    }
147
148    /// Computes degree distributions from a specific code given by polynomials.
149    ///
150    /// # Arguments
151    /// * `r` - Block length (modulus for polynomial arithmetic)
152    /// * `blocks` - A slice of vectors, where each vector contains the
153    ///   coordinates corresponding to the support of the polynomials `h_i` for
154    ///   each block `i`.
155    ///
156    /// # Returns
157    /// A `CodeDegrees` instance containing the computed degree distributions
158    /// for all blocks.
159    pub fn compute_from_code(r: usize, blocks: &[Vec<usize>]) -> CodeDegrees {
160        let mut degree_distributions = Vec::new();
161
162        for block in blocks {
163            // Calculate the square (in Z[x]) of each block
164            let square: HashMap<usize, usize> = block
165                .iter()
166                .flat_map(|&vj| block.iter().map(move |&vk| (vj + vk) % r))
167                .fold(HashMap::new(), |mut map, sum| {
168                    *map.entry(sum).or_insert(0) += 1;
169                    map
170                });
171
172            // Each component of the square is a degree in the subgraph induced by the near
173            // codeword of this block
174            let mut histogram: HashMap<usize, usize> =
175                square.values().fold(HashMap::new(), |mut map, &count| {
176                    *map.entry(count).or_insert(0) += 1;
177                    map
178                });
179
180            // Add count for degree 0
181            if let Some(missing) = r.checked_sub(square.len()) {
182                if missing > 0 {
183                    histogram.insert(0, missing);
184                }
185            }
186
187            degree_distributions.push(histogram);
188        }
189
190        CodeDegrees {
191            degrees: degree_distributions,
192        }
193    }
194}
