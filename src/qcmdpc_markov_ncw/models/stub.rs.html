24/// State of an MDPC code decoding process using syndrome weight (`s`), error
25/// weight (`t`), and the number of common bits with the nearest near codeword
26/// (`u`), along with a block index (`b`).
27#[derive(Clone, Copy, PartialEq, Eq, Hash, Debug, Serialize, Deserialize)]
28pub enum STUBState {
29    /// Valid state with syndrome weight `s`, error weight `t`, and common bits
30    /// `u`.
31    STUB {
32        s: usize,
33        t: usize,
34        u: usize,
35        b: usize,
36    },
37    /// Blocked or failed decoding state.
38    Blocked,
39    NearCodeword,
40    Success,
41}
42
43impl PartialOrd for STUBState {
44    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
45        Some(self.cmp(other))
46    }
47}
48
49impl Ord for STUBState {
50    fn cmp(&self, other: &Self) -> Ordering {
51        match (self, other) {
52            (
53                STUBState::STUB {
54                    s: s1,
55                    t: t1,
56                    u: u1,
57                    b: b1,
58                },
59                STUBState::STUB {
60                    s: s2,
61                    t: t2,
62                    u: u2,
63                    b: b2,
64                },
65            ) => s1
66                .cmp(s2)
67                .then_with(|| t1.cmp(t2))
68                .then_with(|| u1.cmp(u2))
69                .then_with(|| b1.cmp(b2)),
70            (STUBState::Blocked, STUBState::Blocked) => Ordering::Equal,
71            (STUBState::NearCodeword, STUBState::NearCodeword) => Ordering::Equal,
72            (STUBState::Success, STUBState::Success) => Ordering::Equal,
73            (STUBState::Blocked, _) => Ordering::Greater,
74            (_, STUBState::Blocked) => Ordering::Less,
75            (STUBState::NearCodeword, _) => Ordering::Greater,
76            (_, STUBState::NearCodeword) => Ordering::Less,
77            (STUBState::Success, _) => Ordering::Greater,
78            (_, STUBState::Success) => Ordering::Less,
79        }
80    }
81}

83impl State for STUBState {
84    type Basic = STUBasic;
85
86    fn decoded_state() -> Self {
87        STUBState::Success
88    }
89
90    fn to_blocked(&self) -> Self {
91        match self {
92            STUBState::STUB { .. } => STUBState::Blocked,
93            STUBState::Blocked => STUBState::Blocked,
94            STUBState::Success => STUBState::Blocked,
95            STUBState::NearCodeword => STUBState::Blocked,
96        }
97    }
98
99    fn s(&self) -> Result<usize> {
100        match self {
101            STUBState::STUB { s, .. } => Ok(*s),
102            STUBState::Blocked | STUBState::NearCodeword | STUBState::Success => Err(Error::model(
103                "Cannot access syndrome weight on absorbing state",
104            )),
105        }
106    }
107
108    fn t(&self) -> Result<usize> {
109        match self {
110            STUBState::STUB { t, .. } => Ok(*t),
111            STUBState::Blocked | STUBState::NearCodeword | STUBState::Success => Err(Error::model(
112                "Cannot access error weight on absorbing state",
113            )),
114        }
115    }
116
117    fn is_absorbing(&self) -> bool {
118        matches!(
119            self,
120            STUBState::Success | STUBState::Blocked | STUBState::NearCodeword
121        )
122    }
123
124    fn is_success(&self) -> bool {
125        matches!(self, STUBState::Success)
126    }
127}

177/// STUB model using syndrome weight (`s`), error weight (`t`), shared bits
178/// (`u`), and block (`b`) states.
179///
180/// This is the refined model that tracks the decoder state using syndrome
181/// weight, error weight, the number of common bits with the nearest near
182/// codeword, and a block index. It provides more precise predictions than the
183/// ST model by accounting for near codewords that can cause decoding failures
184/// (error floor phenomenon).
185pub struct STUBModel<C: CounterModel<Counter = CountersSTU, BasicState = STUBasic>> {
186    code: MDPCCode,
187    counter_model: C,
188    t_pass: usize,
189    t_fail: usize,
190}
191
192impl<C: CounterModel<Counter = CountersSTU, State = STUBState, BasicState = STUBasic>>
193    STUBModel<C>
194{
195    /// Creates a new STUB model instance.
196    ///
197    /// # Arguments
198    /// * `code` - Reference to the MDPC code structure
199    /// * `counter_model` - Counter model implementing the `CounterModel` trait
200    /// * `t_pass` - Lower error weight threshold (decoder always succeeds below
201    ///   this)
202    /// * `t_fail` - Upper error weight threshold (decoder always fails above
203    ///   this)
204    ///
205    /// # Returns
206    /// A new `STUBModel` instance
207    pub fn new(code: &MDPCCode, counter_model: C, t_pass: usize, t_fail: usize) -> Self {
208        STUBModel {
209            code: code.clone(),
210            counter_model,
211            t_pass,
212            t_fail,
213        }
214    }
215
216    fn get_counters_distribution(&self, state: &STUBState) -> Result<CountersSTU> {
217        match state {
218            STUBState::STUB { .. } => self.counter_model.get_counters_distribution(state),
219            _ => Err(Error::model(format!(
220                "This state cannot have counters: {:?}",
221                state
222            ))),
223        }
224    }
225
226    fn special_state(&self, state: &STUBState) -> Option<STUBState> {
227        match state {
228            STUBState::STUB { s, t, u, .. } => match (*s, *t, *u) {
229                (s, t, u) if s == self.code.d && t == self.code.d && u == self.code.d => {
230                    Some(STUBState::NearCodeword)
231                }
232                (_, t, _) if t >= self.t_fail => Some(STUBState::Blocked),
233                (_, t, _) if t < self.t_pass => Some(STUBState::Success),
234                (s, t, _) if t <= self.code.d && s < t * (self.code.d - t + 1) => {
235                    Some(STUBState::Success)
236                }
237                (s, t, _) if s == 0 && t < 2 * self.code.d => Some(STUBState::Success),
238                _ => None,
239            },
240            _ => None,
241        }
242    }
243
244    fn condition_possible_counters(
245        &self,
246        counter: &CountersSTU,
247        state: &STUBState,
248    ) -> Result<CountersSTU> {
249        match state {
250            STUBState::STUB { s, t, u, .. } => {
251                let CountersSTU {
252                    good,
253                    sus,
254                    err,
255                    bad,
256                } = counter;
257
258                let condition = |counters: &[F64Log], t_dest: usize, u_dest: usize| {
259                    if t_dest != 0 && u_dest == 0 {
260                        return vec![F64Log::new(0.); self.code.d + 1];
261                    }
262
263                    counters
264                        .iter()
265                        .enumerate()
266                        .map(|(sigma, &prob)| {
267                            let s_dest = (*s + self.code.d) as isize - 2 * sigma as isize;
268                            let lower_bound = self.code.d as isize
269                                * (2 * u_dest as isize - t_dest as isize)
270                                - u_dest as isize * (u_dest as isize - 1);
271                            let upper_bound = self.code.d as isize * t_dest as isize
272                                - u_dest as isize * (u_dest as isize - 1);
273
274                            if sigma > *s
275                                || s_dest < 0
276                                || s_dest < lower_bound
277                                || s_dest > upper_bound
278                            {
279                                F64Log::new(0.)
280                            } else {
281                                prob
282                            }
283                        })
284                        .collect::<Vec<_>>()
285                };
286
287                let new_good = condition(good, t + 1, *u).normalize();
288                let new_sus = condition(sus, t + 1, u + 1).normalize();
289                let new_err = if *t > 0 {
290                    condition(err, t - 1, *u).normalize()
291                } else {
292                    None
293                };
294                let new_bad = if *t > 0 && *u > 0 {
295                    condition(bad, t - 1, u - 1).normalize()
296                } else {
297                    None
298                };
299
300                Ok(CountersSTU {
301                    good: new_good.unwrap_or(vec![F64Log::new(0.); self.code.d + 1]),
302                    sus: new_sus.unwrap_or(vec![F64Log::new(0.); self.code.d + 1]),
303                    err: new_err.unwrap_or(vec![F64Log::new(0.); self.code.d + 1]),
304                    bad: new_bad.unwrap_or(vec![F64Log::new(0.); self.code.d + 1]),
305                })
306            }
307            _ => Err(Error::model(format!(
308                "This state cannot have counters: {:?}",
309                state
310            ))),
311        }
312    }
313}

315impl<C: CounterModel<Counter = CountersSTU, State = STUBState, BasicState = STUBasic> + Sync + Send>
316    TransitionModel<STUBState> for STUBModel<C>
317{
318    fn iter_all_states(&self) -> Vec<(usize, Vec<STUBState>)> {
319        (0..=self.code.d * self.t_fail)
320            .map(move |s| {
321                (
322                    s,
323                    (self.t_pass..=self.t_fail)
324                        .flat_map(move |t| {
325                            let mut states = vec![];
326                            for u in 0..=std::cmp::min(t, self.code.d) {
327                                for b in 0..self.code.index {
328                                    if self.code.d * t % 2 != s % 2
329                                        || s > self.code.d * t - u * u.saturating_sub(1)
330                                    {
331                                        continue;
332                                    }
333                                    if u == self.code.d && t == self.code.d && s != self.code.d {
334                                        continue;
335                                    }
336                                    if t == 1 && s != self.code.d {
337                                        continue;
338                                    }
339                                    states.push(STUBState::STUB { s, t, u, b });
340                                }
341                            }
342                            states
343                        })
344                        .collect::<Vec<_>>(),
345                )
346            })
347            .collect()
348    }
349
350    fn transitions_from(
351        &self,
352        state: &STUBState,
353        thresholds: Vec<usize>,
354    ) -> Result<Vec<(STUBState, F64Log)>> {
355        // Handle absorbing states
356        if state.is_absorbing() {
357            return Ok(vec![(*state, F64Log::new(1.0))]);
358        }
359
360        // Handle special states
361        if let Some(special_state) = self.special_state(state) {
362            return Ok(vec![(special_state, F64Log::new(1.0))]);
363        }
364
365        let threshold = thresholds[0];
366
367        match *state {
368            STUBState::STUB { s, t, u, b } => {
369                let counters = self.get_counters_distribution(state)?;
370                let counters = self.condition_possible_counters(&counters, state)?;
371
372                let mut transitions = Vec::new();
373
374                let (p_lock, p_nonlock) =
375                    counters.lock(&self.code, STUBasic { s, t, u }, threshold);
376
377                let uniform_counters = counters.uniform(&self.code, STUBasic { s, t, u })?;
378
379                let (filtered_counters, p_flip) = uniform_counters.filter(threshold);
380                if !p_flip.is_zero() {
381                    let CountersSTU {
382                        good,
383                        sus,
384                        err,
385                        bad,
386                    } = filtered_counters;
387
388                    let process_counter = |counters: &[F64Log], next_t: usize, next_u: usize| {
389                        counters
390                            .iter()
391                            .enumerate()
392                            .filter_map(|(c, &proba_counter)| {
393                                if !proba_counter.is_zero() {
394                                    let next_s = s + self.code.d - 2 * c;
395                                    let potential_state = STUBState::STUB {
396                                        s: next_s,
397                                        t: next_t,
398                                        u: next_u,
399                                        b,
400                                    };
401                                    let next_state = self
402                                        .special_state(&potential_state)
403                                        .unwrap_or(potential_state);
404                                    let probability = proba_counter * p_nonlock / p_flip;
405                                    Some((next_state, probability))
406                                } else {
407                                    None
408                                }
409                            })
410                            .collect::<Vec<_>>()
411                    };
412
413                    transitions.extend(process_counter(&good, t + 1, u));
414                    transitions.extend(process_counter(&sus, t + 1, u + 1));
415                    if t > 0 {
416                        transitions.extend(process_counter(&err, t - 1, u));
417                    }
418                    if t > 0 && u > 0 {
419                        transitions.extend(process_counter(&bad, t - 1, u - 1));
420                    }
421                }
422
423                transitions.push((STUBState::Blocked, p_lock));
424
425                // Merge transitions to the same state
426                Ok(transitions
427                    .into_iter()
428                    .fold(
429                        HashMap::<STUBState, Vec<F64Log>>::new(),
430                        |mut acc, (state, prob)| {
431                            acc.entry(state).or_default().push(prob);
432                            acc
433                        },
434                    )
435                    .into_iter()
436                    .map(|(state, probs)| (state, probs.iter().copied().sum()))
437                    .collect())
438            }
439            _ => Ok(vec![(STUBState::Blocked, F64Log::new(1.0))]),
440        }
441    }
442}

444/// Initial states distribution model for STUB states
445pub struct STUBSpecificInitialStateModel {
446    code: MDPCCode,
447    basic_models: Vec<STUBasicSpecificInitialStateModel>,
448    hypergeom_cache: DashMap<(usize, usize, usize), Vec<F64Log>>,
449}
450
451impl STUBSpecificInitialStateModel {
452    /// Creates a new STUB specific initial state model instance.
453    ///
454    /// # Arguments
455    /// * `code` - Reference to the MDPC code structure
456    /// * `degrees` - Code degrees containing degree distributions for each
457    ///   block
458    ///
459    /// # Returns
460    /// A new `STUBSpecificInitialStateModel` instance
461    pub fn new(code: &MDPCCode, degrees: CodeDegrees) -> Self {
462        STUBSpecificInitialStateModel {
463            code: code.clone(),
464            basic_models: degrees
465                .degrees
466                .iter()
467                .map(|degs| STUBasicSpecificInitialStateModel::new(code, degs.clone()))
468                .collect(),
469            hypergeom_cache: DashMap::new(),
470        }
471    }
472
473    /// Computes the rho values for given parameters
474    ///
475    /// Computes hypergeometric distribution with caching.
476    ///
477    /// # Arguments
478    /// * `n` - Population size (code length)
479    /// * `w` - Number of success states in population (row weight of
480    ///   parity-check matrix)
481    /// * `t` - Number of draws (error weight)
482    ///
483    /// # Returns
484    /// Vector of hypergeometric probabilities
485    pub fn hypergeom(&self, n: usize, w: usize, t: usize) -> Vec<F64Log> {
486        let cache = &self.hypergeom_cache;
487        cache
488            .entry((n, w, t))
489            .or_insert_with(|| F64Log::hypergeom_distribution(n, w, t))
490            .clone()
491    }
492
493    /// Splits initial distribution by the value of u parameter
494    ///
495    /// # Arguments
496    /// * `dist` - HashMap of states and their probabilities
497    ///
498    /// # Returns
499    /// HashMap mapping u values to distributions of states with that u value
500    pub fn split_initial_distribution(
501        dist: HashMap<STUBState, f64>,
502    ) -> HashMap<usize, HashMap<STUBState, f64>> {
503        dist.iter()
504            .fold(HashMap::new(), |mut acc, (&state, &proba)| {
505                if let STUBState::STUB {
506                    s: _,
507                    t: _,
508                    u,
509                    b: _,
510                } = state
511                {
512                    acc.entry(u).or_default().insert(state, proba);
513                };
514                acc
515            })
516    }
517}

519impl InitialStateModel for STUBSpecificInitialStateModel {
520    type State = STUBState;
521
522    fn get_initial_distribution(&self, t: usize) -> HashMap<Self::State, F64Log> {
523        // Calculate the distribution of u (number of common bits with the nearest near
524        // codeword)
525        let u_distr: HashMap<(usize, usize), F64Log> = {
526            // Compute probability distribution for t0
527            let prob_t0: Vec<F64Log> = self.hypergeom(self.code.n, self.code.r, t);
528
529            // Compute u0 for block 0 and u1 for block 1, then find max(u0,u1) and
530            // argmax(u0,u1) We are only interested in the maximum value and the
531            // block in which it occurs This gives us the joint distribution of
532            // max(u0,u1) and argmax(u0,u1)
533            let mut dist = HashMap::new();
534            for u in 0..=self.code.d {
535                for t0 in 0..=t {
536                    let t1 = t - t0;
537                    // Compute CDFs for the maximum of u0 and u1
538                    let f0: Vec<F64Log> = F64Log::max_n_cdf(
539                        &self.hypergeom(self.code.r, self.code.d, t0),
540                        self.code.r,
541                    );
542                    let f1: Vec<F64Log> = F64Log::max_n_cdf(
543                        &self.hypergeom(self.code.r, self.code.d, t1),
544                        self.code.r,
545                    );
546                    // Compute CDFs for the maximum of u0 and u1
547                    let p0: Vec<F64Log> = F64Log::max_n_pmf(
548                        &self.hypergeom(self.code.r, self.code.d, t0),
549                        self.code.r,
550                    );
551                    let p1: Vec<F64Log> = F64Log::max_n_pmf(
552                        &self.hypergeom(self.code.r, self.code.d, t1),
553                        self.code.r,
554                    );
555
556                    // Case 1: u0 = u and u1 < u
557                    let case1_prob = if u > 0 {
558                        p0.get(u).copied().unwrap_or_default()
559                            * f1.get(u - 1).copied().unwrap_or_default()
560                            * prob_t0.get(t0).copied().unwrap_or_default()
561                    } else {
562                        F64Log::new(0.)
563                    };
564
565                    // Case 2: u1 = u and u0 < u
566                    let case2_prob = if u > 0 {
567                        p1.get(u).copied().unwrap_or_default()
568                            * f0.get(u - 1).copied().unwrap_or_default()
569                            * prob_t0.get(t0).copied().unwrap_or_default()
570                    } else {
571                        F64Log::new(0.)
572                    };
573
574                    // Case 3: u0 = u1 = u
575                    let case3_prob: F64Log = p0.get(u).copied().unwrap_or_default()
576                        * p1.get(u).copied().unwrap_or_default()
577                        * prob_t0.get(t0).copied().unwrap_or_default();
578
579                    // Update distribution for argmax = 0 (u0 wins or tie)
580                    *dist.entry((u, 0)).or_insert(F64Log::new(0.)) += case1_prob + case3_prob / 2.;
581
582                    // Update distribution for argmax = 1 (u1 wins or tie)
583                    *dist.entry((u, 1)).or_insert(F64Log::new(0.)) += case2_prob + case3_prob / 2.;
584                }
585            }
586            dist
587        };
588
589        self.basic_models
590            .iter()
591            .enumerate()
592            .flat_map(|(b, model)| {
593                let u_distr = u_distr.clone();
594                model.get_initial_distribution(t).into_iter().map(
595                    move |(STUBasic { s, u, .. }, p_s)| {
596                        let p_u = u_distr.get(&(u, b)).copied().unwrap_or_default();
597                        (STUBState::STUB { s, t, u, b }, p_s * p_u)
598                    },
599                )
600            })
601            .collect()
602    }
