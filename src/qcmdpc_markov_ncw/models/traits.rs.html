6/// Transitions between states in the Markov chain.
7pub trait TransitionModel<S: State>: Send + Sync {
8    /// Generates an iterator of possible states for t between t_low and t_high
9    /// inclusive.
10    fn iter_all_states(&self) -> Vec<(usize, Vec<S>)>;
11
12    /// Computes the possible next states and their transition probabilities
13    /// from a given state
14    ///
15    /// # Arguments
16    /// * `state` - The current state to transition from
17    /// * `threshold` - The threshold value for transitions
18    ///
19    /// # Returns
20    /// A Result containing a vector of tuples with possible next states and
21    /// their transition probabilities
22    fn transitions_from(&self, state: &S, threshold: Vec<usize>) -> Result<Vec<(S, F64Log)>>;
23}

25/// Defines an interface for counter-based models in MDPC decoding.
26/// Provides methods for calculating counter probability distributions
27/// based on decoder states.
28pub trait CounterModel {
29    /// The state type associated with this model.
30    type State: Clone;
31    type BasicState: Clone;
32    /// The counter type associated with this model.
33    type Counter: Counter<Self::BasicState>;
34
35    /// Gets the probability distribution of counters for a state.
36    ///
37    /// Calculates counter distributions needed for the decoding process
38    /// at the current decoder state.
39    ///
40    /// # Arguments
41    ///
42    /// * `state` - A reference to the decoder state
43    ///
44    /// # Returns
45    ///
46    /// Counter probability distribution for the given state, or an error
47    /// if the state cannot have counters.
48    fn get_counters_distribution(&self, state: &Self::State) -> Result<Self::Counter>;
49}

71/// State in the MDPC decoding process.
72///
73/// This trait defines the common interface for different types of states used
74/// in the MDPC code decoding algorithm. Implementations of this trait should
75/// provide methods for calculating counters, determining decoded and blocked
76/// states, and generating possible states from syndrome and error weights.
77///
78/// Every state must be able to convert to a basic representation (via
79/// `to_basic()`) . This basic representation provides the essential information
80/// needed for computation in the Markov chain.
81pub trait State: Clone + Eq + Hash {
82    type Basic: Clone;
83
84    /// Returns the decoded state.
85    fn decoded_state() -> Self;
86
87    /// Returns a blocked state based on the given state.
88    fn to_blocked(&self) -> Self;
89
90    /// Returns the syndrome weight of the current state.
91    ///
92    /// # Errors
93    /// Returns an error if called on a state that does not have a syndrome
94    /// weight.
95    fn s(&self) -> Result<usize>;
96
97    /// Returns the error weight of the current state.
98    ///
99    /// # Errors
100    /// Returns an error if called on a state that does not have an error
101    /// weight.
102    fn t(&self) -> Result<usize>;
103
104    /// Returns whether this state is absorbing in the Markov chain.
105    fn is_absorbing(&self) -> bool;
106
107    /// Returns whether this state is a success in the Markov chain.
108    fn is_success(&self) -> bool;
109}

111/// Counters associated with states in MDPC code decoding.
112///
113/// This trait defines methods for manipulating and analyzing counters that are
114/// used in the decoding process of MDPC codes. It provides functionality for
115/// conditioning counters, computing locking probabilities, applying sampling,
116/// and computing transitions.
117///
118/// # Type Parameters
119///
120/// * `T`: A tuple type representing the state associated with these counters.
121pub trait Counter<T: Clone>: Sync {
122    /// Computes the "locking" probability for the given state and threshold.
123    ///
124    /// # Arguments
125    /// * `code` - A reference to the MDPCCode.
126    /// * `state` - The current state as a tuple.
127    /// * `threshold` - The threshold value for decoding.
128    ///
129    /// # Returns
130    /// The computed locking probability as a F64Log.
131    fn lock(&self, code: &MDPCCode, state: T, threshold: usize) -> (F64Log, F64Log);
132
133    /// Modifies the counters based on uniform sampling.
134    ///
135    /// # Arguments
136    /// * `code` - A reference to the MDPCCode.
137    /// * `state` - The current state as a tuple.
138    ///
139    /// # Returns
140    /// A Result containing either the modified counters or an error message.
141    fn uniform(&self, code: &MDPCCode, state: T) -> Result<Self>
142    where
143        Self: Sized;
144
145    /// Computes transitions for the counters based on the given threshold.
146    ///
147    /// # Arguments
148    /// * `threshold` - The threshold value for transitions.
149    ///
150    /// # Returns
151    /// A tuple containing the new counters, flip probability, and no-flip
152    /// probability.
153    fn filter(&self, threshold: usize) -> (Self, F64Log)
154    where
155        Self: Sized;
156}
