29/// Basic counter model for STU states with specific degree distributions.
30///
31/// This model computes counter distributions for positions in the codeword
32/// based on the current decoder state and specific degree distributions
33/// for near codewords in the Tanner graph.
34pub struct STUBasicSpecificCounterModel {
35    code: MDPCCode,
36    degrees: BlockDegrees,
37    hypergeom_cache: DashMap<(usize, usize, usize), Vec<F64Log>>,
38    probabilities_cache: DashMap<(usize, usize), Vec<F64Log>>,
39}
40
41impl STUBasicSpecificCounterModel {
42    /// Creates a new STU basic specific counter model instance.
43    ///
44    /// # Arguments
45    /// * `code` - Reference to the MDPC code structure
46    /// * `degrees` - Block degrees containing degree distributions for the
47    ///   specific block
48    ///
49    /// # Returns
50    /// A new `STUBasicSpecificCounterModel` instance
51    pub fn new(code: &MDPCCode, degrees: BlockDegrees) -> Self {
52        STUBasicSpecificCounterModel {
53            code: code.clone(),
54            degrees,
55            hypergeom_cache: DashMap::new(),
56            probabilities_cache: DashMap::new(),
57        }
58    }
59
60    fn hypergeom(&self, n: usize, w: usize, t: usize) -> Vec<F64Log> {
61        let cache = &self.hypergeom_cache;
62        cache
63            .entry((n, w, t))
64            .or_insert_with(|| F64Log::hypergeom_distribution(n, w, t))
65            .clone()
66    }
67
68    /// Computes the probabilities for the STU model
69    ///
70    /// # Arguments
71    ///
72    /// * `t` - Error weight
73    /// * `u` - Number of common bits with the nearest near codeword
74    ///
75    /// # Returns
76    ///
77    /// A vector of normalized probabilities
78    ///
79    /// # Formulas
80    ///
81    /// The probabilities computed correspond to the following rho and pi values
82    /// (see Lemmas 6.2-6.7 of \[ALMPRTV25] for details):
83    ///
84    /// Rho values:
85    /// * `ρ_b1(Δ, ℓ)`: Probability that a parity-check labeled `2a` with degree
86    ///   `Δ` in `G` is adjacent to `ℓ` bits in error
87    /// * `ρ_b2(Δ, ℓ)`: Probability that a parity-check labeled `a+b` with
88    ///   degree `Δ` in `G` is adjacent to `ℓ` bits in error, where `b` is
89    ///   another bad bit
90    /// * `ρ_b3(Δ, ℓ)`: Probability that a parity-check labeled `a+b` with
91    ///   degree `Δ` in `G` is adjacent to `ℓ` bits in error, where `b` is a
92    ///   suspicious bit
93    /// * `ρ_s1(Δ, ℓ)`: Probability that a parity-check labeled `2a` with degree
94    ///   `Δ` in `G` is adjacent to `ℓ` bits in error
95    /// * `ρ_s2(Δ, ℓ)`: Probability that a parity-check labeled `a+b` with
96    ///   degree `Δ` in `G` is adjacent to `ℓ` bits in error, where `b` is
97    ///   another suspicious bit
98    /// * `ρ_s3(Δ, ℓ)`: Probability that a parity-check labeled `a+b` with
99    ///   degree `Δ` in `G` is adjacent to `ℓ` bits in error, where `b` is a bad
100    ///   bit
101    /// * `ρ_e(Δ, ℓ)`: Probability that a parity-check node of `G` of degree `Δ`
102    ///   adjacent to a normal bit in error contains exactly `ℓ` erroneous bits
103    /// * `ρ_g(Δ, ℓ)`: Probability that a parity-check node of `G` of degree `Δ`
104    ///   adjacent to a good bit contains exactly `ℓ` erroneous bits
105    ///
106    /// π values:
107    /// * `π_b1 = (1/d) * Σ[Δ odd] n_Δ * Σ[ℓ] ρ_b1(Δ, 2ℓ+1)`
108    /// * `π_b2 = (1/(d(d-1))) * Σ[Δ ≥ 1] (Δ - Δ mod 2)n_Δ * Σ[ℓ] ρ_b2(Δ, 2ℓ+1)`
109    /// * `π_b3 = (1/(d(d-1))) * Σ[Δ ≥ 1] (Δ - Δ mod 2)n_Δ * Σ[ℓ] ρ_b3(Δ, 2ℓ+1)`
110    /// * `π_s1 = (1/d) * Σ[Δ odd] n_Δ * Σ[ℓ] ρ_s1(Δ, 2ℓ+1)`
111    /// * `π_s2 = (1/(d(d-1))) * Σ[Δ ≥ 1] (Δ - Δ mod 2)n_Δ * Σ[ℓ] ρ_s2(Δ, 2ℓ+1)`
112    /// * `π_s3 = (1/(d(d-1))) * Σ[Δ ≥ 1] (Δ - Δ mod 2)n_Δ * Σ[ℓ] ρ_s3(Δ, 2ℓ+1)`
113    /// * `π_e = (1/(d*(n-d))) * Σ[Δ] (w-Δ)n_Δ * Σ[ℓ] ρ_e(Δ, 2ℓ+1)`
114    /// * `π_g = (1/(d*(n-d))) * Σ[Δ] (w-Δ)n_Δ * Σ[ℓ] ρ_g(Δ, 2ℓ+1)`
115    ///
116    /// Where `n_Δ` is the number of parity-check equations of degree `Δ` in
117    /// `G`.
118    ///
119    /// All π values are then normalized by dividing by `E(s|t,u)`, where:
120    /// `E(s|t,u) = (1/w)(u*π_b1 + u(u-1)π_b2 + u(d-u)π_b3 + (d-u)π_s1 +
121    /// (d-u)(d-u-1)π_s2 + (d-u)u*π_s3 + (t-u)d*π_e + (n+u-t-d)d*π_g)`
122    pub fn compute_probabilities(&self, t: usize, u: usize) -> Vec<F64Log> {
123        if let Some(cached) = self.probabilities_cache.get(&(t, u)) {
124            return cached.clone();
125        }
126
127        let d = self.code.d;
128        let w = self.code.w;
129        let n = self.code.n;
130        let d_minus_1 = d.saturating_sub(1);
131        let d_minus_u_minus_1 = d.saturating_sub(u + 1);
132        let u_minus_1 = u.saturating_sub(1);
133        let d_minus_u = d.saturating_sub(u);
134        let t_minus_u = t.saturating_sub(u);
135        let n_minus_d = n.saturating_sub(d);
136        let n_plus_u_minus_t_minus_d = (n + u).saturating_sub(t + d);
137
138        let compute_rho = |deg_ncw, deg_err_ncw, deg_other, deg_err_other, delta, ell| {
139            if deg_ncw <= d
140                && deg_ncw <= delta
141                && deg_err_ncw <= u
142                && deg_err_other + u <= t
143                && deg_err_ncw <= ell
144                && u + deg_ncw <= d + deg_err_ncw
145            {
146                let coeffs1 = self.hypergeom(d - deg_ncw, delta - deg_ncw, u - deg_err_ncw);
147                let coeffs2 = self.hypergeom(
148                    n - d - deg_other,
149                    w - delta - deg_other,
150                    t - u - deg_err_other,
151                );
152                (deg_err_other..=ell - deg_err_ncw)
153                    .map(|j| {
154                        *coeffs1.get(j - deg_err_other).unwrap_or(&F64Log::new(0.))
155                            * *coeffs2
156                                .get(ell - deg_err_ncw - j)
157                                .unwrap_or(&F64Log::new(0.))
158                    })
159                    .sum()
160            } else {
161                F64Log::new(0.)
162            }
163        };
164
165        let compute_sum = |degrees: &HashMap<usize, usize>,
166                           deg_ncw,
167                           deg_err_ncw,
168                           deg_other,
169                           deg_err_other| {
170            degrees
171                .iter()
172                .map(|(delta, &coeff)| {
173                    let odd_ell_sum: F64Log = (1..=w)
174                        .step_by(2)
175                        .map(|ell| {
176                            compute_rho(deg_ncw, deg_err_ncw, deg_other, deg_err_other, *delta, ell)
177                        })
178                        .sum();
179                    odd_ell_sum * coeff as f64
180                })
181                .sum::<F64Log>()
182        };
183
184        // The following values correspond to factors in the computation of π:
185        //
186        // 1. double_a: corresponds to `(Δ mod 2) n_Δ`
187        //
188        // 2. a_plus_b: corresponds to `(Δ - Δ mod 2) n_Δ`
189        //
190        // 3. other: corresponds to `(w - Δ) n_Δ`
191        let double_a: HashMap<usize, usize> = self
192            .degrees
193            .iter()
194            .filter(|&(&degree, _)| degree % 2 != 0)
195            .map(|(&degree, &count)| (degree, count))
196            .collect();
197
198        let a_plus_b: HashMap<usize, usize> = self
199            .degrees
200            .iter()
201            .filter(|&(&degree, _)| degree > 1)
202            .map(|(&degree, &count)| (degree, count * (degree - (degree % 2))))
203            .collect();
204
205        let other: HashMap<usize, usize> = self
206            .degrees
207            .iter()
208            .map(|(&degree, &count)| (degree, count * (w - degree)))
209            .collect();
210
211        let sum_b1: F64Log = compute_sum(&double_a, 1, 1, 0, 0) / d as f64;
212        let sum_b2: F64Log = compute_sum(&a_plus_b, 2, 2, 0, 0) / (d * d_minus_1) as f64;
213        let sum_b3: F64Log = compute_sum(&a_plus_b, 2, 1, 0, 0) / (d * d_minus_1) as f64;
214        let sum_s1: F64Log = compute_sum(&double_a, 1, 0, 0, 0) / d as f64;
215        let sum_s2: F64Log = compute_sum(&a_plus_b, 2, 0, 0, 0) / (d * d_minus_1) as f64;
216        let sum_s3: F64Log = compute_sum(&a_plus_b, 2, 1, 0, 0) / (d * d_minus_1) as f64;
217        let sum_e: F64Log = compute_sum(&other, 0, 0, 1, 1) / (n_minus_d * d) as f64;
218        let sum_g: F64Log = compute_sum(&other, 0, 0, 1, 0) / (n_minus_d * d) as f64;
219
220        let weights = [
221            u as f64,
222            (u * u_minus_1) as f64,
223            (u * d_minus_u) as f64,
224            d_minus_u as f64,
225            (d_minus_u * d_minus_u_minus_1) as f64,
226            (d_minus_u * u) as f64,
227            (t_minus_u * d) as f64,
228            (n_plus_u_minus_t_minus_d * d) as f64,
229        ];
230
231        let probabilities = [sum_b1, sum_b2, sum_b3, sum_s1, sum_s2, sum_s3, sum_e, sum_g];
232
233        let s_knowing_t_u = weights
234            .iter()
235            .zip(probabilities.iter())
236            .map(|(&wt, &pr)| pr * wt)
237            .sum::<F64Log>()
238            / w as f64;
239
240        let normalized_probabilities: Vec<F64Log> =
241            probabilities.iter().map(|&p| p / s_knowing_t_u).collect();
242
243        self.probabilities_cache
244            .insert((t, u), normalized_probabilities.clone());
245        normalized_probabilities
246    }
247}

249impl CounterModel for STUBasicSpecificCounterModel {
250    type State = STUBasic;
251    type BasicState = STUBasic;
252    type Counter = CountersSTU;
253
254    /// Generates a counter for the given state
255    ///
256    /// # Arguments
257    ///
258    /// * `state` - The current state
259    ///
260    /// # Returns
261    ///
262    /// A CountersSTU instance
263    ///
264    /// # Models
265    ///
266    /// This function implements Model 3 of \[ALMPRTV25] for the different types
267    /// of positions:
268    ///
269    /// 1. Bad bit counter: Modeled by the sum of:
270    ///    - A Bernoulli random variable with parameter `s * π_b1 / E(s|t,u)`
271    ///    - (u-1) Bernoulli random variables with parameter `s * π_b2 /
272    ///      E(s|t,u)`
273    ///    - (d-u) Bernoulli random variables with parameter `s * π_b3 /
274    ///      E(s|t,u)`
275    ///
276    /// 2. Suspicious bit counter: Modeled by the sum of:
277    ///    - A Bernoulli random variable with parameter `s * π_s1 / E(s|t,u)`
278    ///    - (d-u-1) Bernoulli random variables with parameter `s * π_s2 /
279    ///      E(s|t,u)`
280    ///    - u Bernoulli random variables with parameter `s * π_s3 / E(s|t,u)`
281    ///
282    /// 3. Normal bit in error counter: Modeled by the sum of d Bernoulli random
283    ///    variables with parameter `s * π_e / E(s|t,u)`
284    ///
285    /// 4. Good bit counter: Modeled by the sum of d Bernoulli random variables
286    ///    with parameter `s * π_g / E(s|t,u)`
287    fn get_counters_distribution(&self, state: &Self::State) -> Result<Self::Counter> {
288        let STUBasic { s, t, u } = state;
289        let normalized_probabilities = self.compute_probabilities(*t, *u);
290
291        let bad = if *u > 0 {
292            let x1 = F64Log::binomial_distribution(normalized_probabilities[0] * *s as f64, 1);
293            let x2 = F64Log::binomial_distribution(normalized_probabilities[1] * *s as f64, *u - 1);
294            let x3 = F64Log::binomial_distribution(
295                normalized_probabilities[2] * *s as f64,
296                self.code.d - *u,
297            );
298            x1.convolve(&x2, *u).convolve(&x3, self.code.d)
299        } else {
300            vec![F64Log::new(0.0); self.code.d + 1]
301        };
302
303        let sus = if *u < self.code.d {
304            let x1 = F64Log::binomial_distribution(normalized_probabilities[3] * *s as f64, 1);
305            let x2 = F64Log::binomial_distribution(
306                normalized_probabilities[4] * *s as f64,
307                self.code.d - *u - 1,
308            );
309            let x3 = F64Log::binomial_distribution(normalized_probabilities[5] * *s as f64, *u);
310            x1.convolve(&x2, self.code.d - *u)
311                .convolve(&x3, self.code.d)
312        } else {
313            vec![F64Log::new(0.0); self.code.d + 1]
314        };
315
316        let err = if *u < *t {
317            F64Log::binomial_distribution(normalized_probabilities[6] * *s as f64, self.code.d)
318        } else {
319            vec![F64Log::new(0.0); self.code.d + 1]
320        };
321
322        let good = if self.code.n + *u >= self.code.d + *t {
323            F64Log::binomial_distribution(normalized_probabilities[7] * *s as f64, self.code.d)
324        } else {
325            vec![F64Log::new(0.0); self.code.d + 1]
326        };
327
328        Ok(CountersSTU {
329            good,
330            sus,
331            err,
332            bad,
333        })
334    }
335}

337/// Basic initial state model for STU states with specific degree distributions.
338///
339/// This model computes the initial probability distribution of states
340/// for a given error weight using specific degree distributions and
341/// hypergeometric distributions.

375impl InitialStateModel for STUBasicSpecificInitialStateModel {
376    type State = STUBasic;
377    /// Generates the initial state distribution for a given error weight.
378    ///
379    /// Details are given in Appendix A of \[ALMPRTV25].
380    ///
381    /// # Arguments
382    ///
383    /// * `t` - error weight for which to compute the initial distribution
384    ///
385    /// # Returns
386    ///
387    /// A HashMap mapping each possible initial state to its probability
388    fn get_initial_distribution(&self, t: usize) -> HashMap<Self::State, F64Log> {
389        let r = self.code.r;
390        let n = self.code.n;
391        let d = self.code.d;
392        let w = self.code.w;
393        let alpha = F64Log::from_ln(-10f64.powf(-ALPHA as f64).ln_1p(), 1);
394
395        (0..=d)
396            .flat_map(|u| {
397                // Helper function to compute rho values
398                let compute_rho = |delta, ell| {
399                    let coeffs1 = self.hypergeom(d, delta, u);
400                    let coeffs2 = self.hypergeom(n - d, w - delta, t - u);
401                    (0..=ell)
402                        .map(|j| {
403                            coeffs1.get(j).copied().unwrap_or(F64Log::new(0.))
404                                * coeffs2.get(ell - j).copied().unwrap_or(F64Log::new(0.))
405                        })
406                        .sum()
407                };
408
409                let degrees = &self.degrees;
410
411                // Compute rho values for each degree
412                let rhos: HashMap<usize, OffsetVec<F64Log>> = degrees
413                    .keys()
414                    .map(|&delta| {
415                        let distribution = (0..=w)
416                            .map(|ell| compute_rho(delta, ell))
417                            .collect::<Vec<F64Log>>();
418                        (
419                            delta,
420                            OffsetVec {
421                                offset: 0,
422                                stride: 1,
423                                vec: distribution,
424                            },
425                        )
426                    })
427                    .collect();
428
429                // Split rhos into odd and even parts
430                let (rhos_odd, rhos_even): (
431                    HashMap<usize, OffsetVec<F64Log>>,
432                    HashMap<usize, OffsetVec<F64Log>>,
433                ) = rhos
434                    .into_iter()
435                    .map(|(delta, rho)| {
436                        let (odd, even) = rho.split_parity();
437                        ((delta, odd.trim_zeros()), (delta, even.trim_zeros()))
438                    })
439                    .unzip();
440
441                // Helper function to compute convolution powers for all possible s <= weight
442                let compute_convolution_powers = |vec: OffsetVec<F64Log>, weight: usize| {
443                    // Compute all convolution powers at once
444                    let convolution_powers = vec.convolve_all_powers(weight, d * t);
445
446                    // Divide each power by its factorial
447                    let convolution_powers: HashMap<_, _> = convolution_powers
448                        .into_par_iter()
449                        .map(|(s, convolved_vec)| {
450                            // Divide each component by s! (fact_s)
451                            // This is done to ensure that when we convolve this vector with others
452                            // later, we're effectively computing the
453                            // exponential generating function of the distribution
454                            let divided_vec = convolved_vec
455                                .vec
456                                .iter()
457                                .map(|component| *component / F64Log::factorial(s))
458                                .collect::<Vec<_>>();
459
460                            (
461                                s,
462                                OffsetVec {
463                                    offset: convolved_vec.offset,
464                                    stride: convolved_vec.stride,
465                                    vec: divided_vec,
466                                },
467                            )
468                        })
469                        .collect();
470                    convolution_powers
471                };
472
473                // Compute convolution powers for odd and even parts of each rho (for each
474                // degree delta)
475                let s_odd: HashMap<usize, _> = rhos_odd
476                    .into_par_iter()
477                    .map(|(delta, s)| {
478                        (
479                            delta,
480                            compute_convolution_powers(s, *degrees.get(&delta).unwrap()),
481                        )
482                    })
483                    .collect();
484
485                let s_even: HashMap<usize, _> = rhos_even
486                    .into_par_iter()
487                    .map(|(delta, s)| {
488                        (
489                            delta,
490                            compute_convolution_powers(s, *degrees.get(&delta).unwrap()),
491                        )
492                    })
493                    .collect();
494
495                // For each degree delta, we convolve s nodes with odd rho and (degrees[delta] -
496                // s) nodes with even rho This ensures we have exactly
497                // degrees[delta] nodes of degree delta in total
498                let s: Vec<_> = s_odd
499                    .into_par_iter()
500                    .map(|(delta, odd)| {
501                        let even = s_even.get(&delta).unwrap();
502                        let weight = { *degrees.get(&delta).unwrap() };
503                        odd.convolve_left(even, weight, d * t).simplify(&alpha)
504                    })
505                    .collect();
506
507                // Iteratively convolve the array of sequence distributions together:
508                // We only need to compute one specific term in the final convolution.
509                // So we split the sequences into the largest one and the rest
510                // Then convolve all the rest together first
511                // And finally use convolve_one() for the final convolution with the largest
512                // sequence This avoids computing unnecessary terms in the final
513                // convolution
514                let (s_largest, s_rest) = {
515                    let mut sorted = s.clone();
516                    sorted.sort_by_key(|m| m.len());
517                    let largest = sorted.pop().unwrap_or_default();
518                    (largest, sorted)
519                };
520
521                let s_sum = s_rest
522                    .into_iter()
523                    .convolve_all(d * t)
524                    .unwrap_or_else(HashMap::new)
525                    .simplify(&alpha);
526
527                // Compute syndrome distribution
528                // This calculation considers only the convolutions of s_sum with s_largest
529                // where the total weight is d*t. d*t represents the total
530                // number of connected edges in the Tanner graph between parity checks and error
531                // bits. The resulting distribution gives probabilities for
532                // different syndrome weight values at this fixed total
533                // weight d*t.
534                let s_dist: Vec<F64Log> = (0..=r)
535                    .into_par_iter()
536                    .map(|s| {
537                        if d * t % 2 == s % 2 && s + u * u.saturating_sub(1) <= d * t {
538                            s_sum.convolve_one(&s_largest, &[s, (d * t)])
539                        } else {
540                            F64Log::new(0.)
541                        }
542                    })
543                    .collect();
544
545                let norm_s_dist = s_dist.normalize().unwrap_or_default();
546
547                // Create STUState for each s and multiply by probability of u
548                norm_s_dist
549                    .into_iter()
550                    .enumerate()
551                    .filter_map(|(s, p)| {
552                        if !p.is_zero() {
553                            Some((STUBasic { s, t, u }, p))
554                        } else {
555                            None
556                        }
557                    })
558                    .collect::<Vec<_>>()
559            })
560            .collect()
561    }
